pipeline {
    agent any
    stages {
        stage('Build Docker Image') {
            steps {
                script {
                    dockerImage = docker.build("autotest-image", "./autotest")
                }
            }
        }
        stage('Run Tests') {
            steps {
                script {
                    dockerImage.run("--name autotest-container -v ${WORKSPACE}/autotest:/autotest", "npm run wdio")
                }
            }
        }
    }
    post {
        always {
            script {
                docker rm -f autotest-container
            }
        }
    }
}
